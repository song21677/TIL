# 스택
* 분류 : 스택
* 난이도 : 실버4

<br><br>

## 문제
[백준 10828번 스택](https://www.acmicpc.net/problem/10828)

<br><br>

## 내 풀이
```java
import java.util.*;
import java.io.*;

class Main {
    static Main hw = new Main();
    BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
    StringTokenizer st;
    Stack<Integer> s = new Stack<>();
    
    public static void main(String[] args) throws IOException {
        int N; String cmd, operation;
        N = Integer.parseInt(hw.br.readLine());
       
        for (int i=1; i<=N; i++) {
            cmd = hw.br.readLine();
            if (cmd.contains(" ")) {
                hw.st = new StringTokenizer(cmd, " ");
                cmd = hw.st.nextToken();
                hw.Execute(operation);
            }
            else hw.Execute(cmd);
        }
    }
    
    public void Execute(String cmd) throws IOException {
        switch(cmd) {
            case "push" :
                push();
                break;
            case "pop" :
                pop();
                break;
            case "size" :
                size();
                break;
            case "empty" :
                empty();
                break;
            default :
                peek();
                break;
        }
    }
    
    public void push() throws IOException{
        int num;
        num = Integer.parseInt(st.hw.nextToken());
        s.push(num);
    }
    
    public void pop() {
        if (s.empty()) System.out.println(-1);
        else System.out.println(s.pop());
    }
    
    public void size() {
        int size = 0;
        for (int a : s) size++;
        System.out.println(size);
    }
    
    public void empty() {
        if (s.empty()) System.out.println(1);
        else System.out.println(0);
    }
    
    public void peek() {
        if (s.empty()) System.out.println(-1);
        else System.out.println(s.peek());
    }
}
```
