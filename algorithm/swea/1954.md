# 달팽이 숫자
* 난이도: D2
* 피드백
  * r과 c를 범위 검사나 방문 체크를 할 때 증가시켜서, 범위에 속하지 않고 방문한 적이 있어도 증가돼 문제가 생겼었다.
* 풀이
  ```java
  import java.util.Scanner;
  
  class Solution {
	public static void main(String args[]) throws Exception
	{
		Scanner sc = new Scanner(System.in);
		int T;
		T=sc.nextInt();

		for(int test_case = 1; test_case <= T; test_case++)
		{
			int n = sc.nextInt();
            int arr [][] = new int [n][n];
            boolean [][] visited = new boolean [n][n];
            
            int r = 0, c = -1;
            int num = 0;
            
            while (num < n*n) {
                while (checkRange(r, c+1, n) && !visited[r][c+1]) {
                    c++;
                    arr[r][c] = ++num;
                    visited[r][c] = true;
                }

                while (checkRange(r+1, c, n) && !visited[r+1][c]) {
                    r++;
                    arr[r][c] = ++num;
                    visited[r][c] = true;
                }

                while (checkRange(r, c-1, n) && !visited[r][c-1]) {
                    c--;
                    arr[r][c] = ++num;
                    visited[r][c] = true;
                }

                while (checkRange(r-1, c, n) && !visited[r-1][c]) {
                    r--;
                    arr[r][c] = ++num;
                    visited[r][c] = true;
                }
            }
            
            System.out.println("#" + test_case);
            for (int i=0; i<n; i++) {
                for (int j=0; j<n; j++) {
                    System.out.print(arr[i][j] + " ");
                }
                System.out.println();
            }
		}
	}
    
    private static boolean checkRange(int r, int c, int N) {
        if (r>=0 && r<=N-1 && c>=0 && c<=N-1) return true;
        return false;
    }
  }
  ```
