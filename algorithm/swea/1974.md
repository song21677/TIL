# 스도쿠 검증
* 난이도: D2
* 알고리즘: 구현
* 풀이
* 5번 케이스 문제,,,
  ```JAVA
  import java.util.Scanner;
  import java.util.Arrays;
  
  class Solution {
	public static void main(String args[]) throws Exception
	{
		Scanner sc = new Scanner(System.in);
		int T;
		T=sc.nextInt();

		for(int test_case = 1; test_case <= T; test_case++)
		{
            System.out.print("#" + test_case + " ");
            
			int[][] arr = new int[9][9];
            boolean[] visited = new boolean[10];
            
            // 입력
            for (int i=0; i<9; i++) {
                for (int j=0; j<9; j++) {
                    arr[i][j] = sc.nextInt();
                }
            }
            
            boolean check = true;
            
             // 행 우선순위 검사(가로)
            for (int i=0; i<9; i++) {
                Arrays.fill(visited, false);
                for (int j=0; j<9; j++) {
                    if (!visited[arr[i][j]]) visited[arr[i][j]] = true;
                    else {
                        j=9; i=9; check = false;
                    }
                }
            }
            
            if (!check) {
                System.out.println(0);
                continue;
            }
            
            // 열 우선순위 검사(세로)
            for (int j=0; j<9; j++) {
                Arrays.fill(visited, false);
                for (int i=0; i<9; i++) {
                    if (!visited[arr[i][j]]) visited[arr[i][j]] = true;
                    else {
                        j=9; i=9; check = false;
                    }
                }
            }
            
            if (!check) {
                System.out.println(0);
                continue;
            }
            
            // 3*3 배열 검사
            int i=0, j=0;
            int r=i+3, c = j+3;
            while (i<9) {
                while (j<9) {
                    r = i+3;
                    c = j+3;
                    Arrays.fill(visited, false);
                    for (; i<r; i++) {
                        for (; j<c; j++) {
                            if (!visited[arr[i][j]]) visited[arr[i][j]] = true;
                            else {
                                j=9; i=9; check = false;
                            }
                        }
                    }
                    j+=3;
                }
                i+=3;
            }
            
            if (!check) System.out.println(0);
            else System.out.println(1);
		}
    }
  }
  ```
